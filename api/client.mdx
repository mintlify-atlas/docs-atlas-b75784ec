## Client Class

The `Client` class provides a low-level REST API client with type-safe HTTP methods and built-in caching support.

```typescript
import { Client } from '@tonightpass/node';

const client = new Client({
  baseURL: 'https://api.tonightpass.com',
  apiKey: 'your-api-key',
  cache: { enabled: true }
});
```

## Methods

### get

Performs a GET request to the specified endpoint.

```typescript
async get<Path extends PathsFor<"GET">>(
  path: Path,
  query?: Query<Path>,
  options?: APIRequestOptions
): Promise<ResponseFor<"GET", Path>>
```

<ParamField path="path" type="PathsFor<'GET'>" required>
  The API endpoint path. Must be a valid GET endpoint from the API
</ParamField>

<ParamField path="query" type="Query<Path>">
  Query parameters as key-value pairs
</ParamField>

<ParamField path="options" type="APIRequestOptions">
  Additional request options (headers, timeout, etc.)
</ParamField>

<ResponseField name="return" type="Promise<ResponseFor<'GET', Path>>">
  Returns the typed response data for the endpoint
</ResponseField>

**Example:**


## TonightPassAPIError

Error class thrown when an API request fails.

```typescript
class TonightPassAPIError<T> extends Error {
  public readonly status: number;
  public readonly response: RedaxiosResponse<APIResponse<T>>;
  public readonly data: ErroredAPIResponse;
}
```

<ResponseField name="message" type="string">
  The error message from the API
</ResponseField>

<ResponseField name="status" type="number">
  HTTP status code
</ResponseField>

<ResponseField name="response" type="RedaxiosResponse<APIResponse<T>>">
  The full HTTP response object
</ResponseField>

<ResponseField name="data" type="ErroredAPIResponse">
  The error response data
  
  <Expandable title="ErroredAPIResponse properties">
    <ResponseField name="success" type="false">
      Always `false` for error responses
    </ResponseField>
    
    <ResponseField name="message" type="string">
      Error message
    </ResponseField>
    
    <ResponseField name="statusCode" type="number">
      HTTP status code
    </ResponseField>
    
    <ResponseField name="errors" type="Record<string, string[]>">
      Validation errors by field name (if applicable)
    </ResponseField>
  </Expandable>
</ResponseField>

**Example:**

```typescript
try {
  await client.get('/organizations/:id', { id: 'invalid' });
} catch (error) {
  if (error instanceof TonightPassAPIError) {
    console.error(`Error ${error.status}: ${error.message}`);
    if (error.data.errors) {
      console.error('Validation errors:', error.data.errors);
    }
  }
}
```

## APIRequestOptions

Additional options that can be passed to HTTP methods.

```typescript
interface APIRequestOptions extends Options {
  apiKey?: string;
  accessToken?: string;
}
```

<ParamField path="apiKey" type="string">
  Override the client's API key for this request
</ParamField>

<ParamField path="accessToken" type="string">
  Override the client's access token for this request
</ParamField>

<ParamField path="headers" type="Record<string, string>">
  Additional HTTP headers
</ParamField>

<ParamField path="timeout" type="number">
  Request timeout in milliseconds
</ParamField>

**Example:**

```typescript
const data = await client.get('/organizations/:id', 
  { id: 'org-123' },
  { 
    headers: { 'X-Custom-Header': 'value' },
    timeout: 5000
  }
);
```
